.item { background-color: aqua; align-items: stretch; width: 40px; }

.item2 { background-color: rgb(29, 199, 199); align-self: center; width: 40px; height: 50px; }

.item3 { background-color: rgb(29, 163, 163); align-self: flex-start; width: 40px; height: 50px; }

.item4 { background-color: rgb(29, 163, 163); align-self: flex-end; width: 40px; height: 50px; }
